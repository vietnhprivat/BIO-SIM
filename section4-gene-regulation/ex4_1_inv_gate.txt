number kon  = 1
number koff = 1
number dA   = 1/10000
number dB   = 1/10
number b    = 1

species {A, Pp, P, B}

amount A  @ 0 M
amount Pp @ 1 M
amount P  @ 0 M
amount B  @ 0 M

A + Pp -> {kon}  P
P      -> {koff} A + Pp

Pp -> {b} Pp + B

A -> {dA} #
B -> {dB} #

trigger A @ 1 M when time > 50

equilibrate for 200
