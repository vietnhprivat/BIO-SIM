number kon  = 1
number koff = 1
number dA   = 1/10000
number dB   = 1/10
number b    = 1

species {A, P, Pp, B}

amount A  @ 0 M
amount P  @ 1 M
amount Pp @ 0 M
amount B  @ 0 M

A + P  -> {kon}  Pp
Pp     -> {koff} A + P

Pp -> {b} Pp + B

A -> {dA} #
B -> {dB} #

trigger A @ 4 M when time > 50
trigger A @ 0 M when time > 200

equilibrate for 400
